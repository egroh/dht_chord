1//! Provides a server socket for API communication
2//!
3//! The API-communications module is responsible for communication with other modules.
4//! Incoming requests are dispatched to a worker thread pool,
5//! that will process each request independently and asynchronously.
6//!
7//! Answer to messages received are not guaranteed to be sent in the same order as the requests were received.
8//!
9//! As per specification, the following requests are accepted and processed:
10//!
11//! - DHT PUT
12//! - DHT GET
13//! - DHT SUCCESS
14//! - DHT FAILURE
15//!
16//!  We introduced an additional API Message, [`API_DHT_SHUTDOWN`] which allows us to shut down a node gracefully through
17//!  the API.
18//!  The package has a fixed size and does not contain any other information except the header:
19//!  ```markdown
20//!  +--------+--------+--------+--------+
21//!  |        4        |  DHT SHUTDOWN   |
22//!  +--------+--------+--------+--------+
23//!  ```
24//!
25//! All [`ApiPacket`] s we receive, are first parsed by their [`ApiPacketHeader`].
26//! Depending on the header,
27//! we then parse the rest of the message into the corresponding [`ApiPacketMessage`].

48/// Requests the DHT to store a value
49pub const API_DHT_PUT: u16 = 650;
50
51/// Requests the DHT to retrieve a value
52pub const API_DHT_GET: u16 = 651;
53
54/// Answer to a successful [`API_DHT_GET`] request
55pub const API_DHT_SUCCESS: u16 = 652;
56
57/// Answer to a failed [`API_DHT_GET`] request
58pub const API_DHT_FAILURE: u16 = 653;
59
60/// Requests our DHT node to shutdown
61///
62/// The corresponding packet consists only of a message-header, with the message-type set to `654`.
63pub const API_DHT_SHUTDOWN: u16 = 654;
64
65/// Internal representation of all packages received on the API socket
66///
67/// After 4 header bytes are received and parsed into an [`ApiPacketHeader`],
68/// the rest of the message is parsed into the corresponding [`ApiPacketMessage`].
69pub struct ApiPacket {
70    header: ApiPacketHeader,
71    message: ApiPacketMessage,
72}
73/// Header of an [`ApiPacket`]
74#[derive(Serialize, Deserialize, Debug)]
75pub struct ApiPacketHeader {
76    /// The size indicates the total length of a message, *including* the header
77    pub(crate) size: u16,
78    /// Indicates the type of the message with a well-known constant
79    pub(crate) message_type: u16,
80}
81
82/// Content of an [`ApiPacket`]
83///
84/// Once enough bytes have been received,
85/// the message is parsed fom `Unparsed` into the matching request.
86pub enum ApiPacketMessage {
87    Put(DhtPut),
88    Get(DhtGet),
89    Shutdown,
90    Unparsed(Vec<u8>),
91}
92
93#[derive(Debug)]
94/// Internal representation of a `DHT_PUT` request
95pub struct DhtPut {
96    /// Time-to-live of the value in seconds
97    pub(crate) ttl: u16,
98    /// How many copies of the value should be stored in the DHT
99    pub(crate) replication: u8,
100    /// Reserved for future use
101    pub(crate) _reserved: u8,
102    /// Fixed size 256 bit key (we hash all keys into a 64 bit integer before storing them)
103    pub(crate) key: [u8; 32],
104    /// Value of arbitrary size
105    pub(crate) value: Vec<u8>,
106}
107
108/// Internal representation of a `DHT_GET` request
109#[derive(Deserialize, Debug)]
110pub struct DhtGet {
111    key: [u8; 32],
112}
113
114/// Internal representation of a `DHT_SUCCESS` response
115#[derive(Debug)]
116pub struct DhtGetSuccess {
117    key: [u8; 32],
118    value: Vec<u8>,
119}
120
121/// Internal representation of a `DHT_FAILURE` response
122#[derive(Serialize, Debug)]
123pub struct DhtGetFailure {
124    key: [u8; 32],
125}

127impl ApiPacket {
128    fn default() -> Self {
129        ApiPacket {
130            header: ApiPacketHeader {
131                size: 0,
132                message_type: 0,
133            },
134            message: ApiPacketMessage::Unparsed(Vec::new()),
135        }
136    }
137    fn parse(&mut self, byte: u8) -> Result<(), Box<dyn Error>> {
138        if let ApiPacketMessage::Unparsed(v) = &mut self.message {
139            v.push(byte);
140            if self.header.size as usize <= v.len() + 4 {
141                match self.header.message_type {
142                    API_DHT_PUT => {
143                        if self.header.size < 4 + 4 + 32 + 1 {
144                            return Err(
145                                format!["DHT PUT invalid size: {}", self.header.size].into()
146                            );
147                        }
148                        let ttl = u16::from_be_bytes([v[0], v[1]]);
149                        let replication = v[2];
150                        let reserved = v[3];
151                        let key = &v[4..36];
152                        let value = v[36..].to_vec();
153                        debug_assert!(value.len() == self.header.size as usize - 40);
154                        self.message = ApiPacketMessage::Put(DhtPut {
155                            ttl,
156                            replication,
157                            _reserved: reserved,
158                            key: key.try_into().unwrap(),
159                            value,
160                        });
161                    }
162                    API_DHT_GET => {
163                        if self.header.size != 4 + 32 {
164                            return Err(
165                                format!["DHT GET invalid size: {}", self.header.size].into()
166                            );
167                        }
168                        self.message = ApiPacketMessage::Get(
169                            DefaultOptions::new()
170                                .with_fixint_encoding()
171                                .with_big_endian()
172                                .deserialize(v)?,
173                        );
174                    }
175                    API_DHT_SHUTDOWN => {
176                        if self.header.size != 4 {
177                            return Err("Invalid size".into());
178                        }
179                        self.message = ApiPacketMessage::Shutdown;
180                    }
181                    API_DHT_FAILURE => {
182                        panic!("We should only send these");
183                    }
184                    API_DHT_SUCCESS => {
185                        panic!("We should only send these");
186                    }
187                    _ => return Err("Unknown message type".into()),
188                }
189            }
190        } else {
191            panic!("Message already parsed");
192        }
193        Ok(())
194    }
195}

197pub(crate) fn hash_key_bytes(vec_bytes: &[u8]) -> u64 {
198    let mut hasher = DefaultHasher::new();
199    vec_bytes.hash(&mut hasher);
200    hasher.finish()
201}
202pub(crate) async fn process_api_put_request(dht: Chord, put: DhtPut) {
203    let hashed_key = hash_key_bytes(&put.key);
204    if let Err(e) = dht
205        .insert(
206            hashed_key,
207            put.value,
208            Duration::from_secs(put.ttl as u64),
209            put.replication,
210        )
211        .await
212    {
213        warn!("Error inserting key {:?} into DHT: {}", &put.key, e);
214    }
215}
216async fn process_api_get_request(
217    dht: Chord,
218    get: &DhtGet,
219    response_stream: &Arc<Mutex<OwnedWriteHalf>>,
220) {
221    let hashed_key = hash_key_bytes(&get.key);
222    match dht.get(hashed_key).await {
223        Some(value) => {
224            let header = ApiPacketHeader {
225                size: 4 + get.key.len() as u16 + value.len() as u16,
226                message_type: API_DHT_SUCCESS,
227            };
228            let mut buf = DefaultOptions::new()
229                .with_fixint_encoding()
230                .with_big_endian()
231                .serialize(&header)
232                .unwrap();
233            buf.extend(get.key);
234            buf.extend(value);
235
236            if let Err(e) = response_stream.lock().await.write_all(&buf).await {
237                warn!("Error writing to socket: {}", e);
238            }
239        }
240        None => {
241            let header = ApiPacketHeader {
242                size: 4 + get.key.len() as u16,
243                message_type: API_DHT_FAILURE,
244            };
245            let mut buf = DefaultOptions::new()
246                .with_fixint_encoding()
247                .with_big_endian()
248                .serialize(&header)
249                .unwrap();
250            buf.extend(get.key);
251
252            if let Err(e) = response_stream.lock().await.write_all(&buf).await {
253                warn!("Error writing to socket: {}", e);
254            }
255        }
256    }
257}
258pub(crate) async fn start_api_server(
259    dht: Chord,
260    api_address: SocketAddr,
261    cancellation_token: CancellationToken,
262) -> JoinHandle<()> {
263    let api_listener = TcpListener::bind(api_address)
264        .await
265        .expect("Failed to bind API server socket");
266
267    // Open channel for inter thread communication
268    let (thread_start_tx, mut thread_start_rx) = mpsc::channel(1);
269
270    info!("Listening for API Calls on {}", api_address);
271    let handle = tokio::spawn(async move {
272        // Send signal that we are running
273        thread_start_tx
274            .send(true)
275            .await
276            .expect("Unable to send message");
277
278        loop {
279            tokio::select! {
280                result = api_listener.accept() => {
281                    if let Ok((stream, _)) = result {
282
283                    let (mut reader, writer) = stream.into_split();
284                    let writer = Arc::new(Mutex::new(writer));
285
286                    let dht = dht.clone();
287                    let cancellation_token_clone = cancellation_token.clone();
288
289                    tokio::spawn(async move {
290                        let connection_result = async move || -> Result<(), Box<dyn Error>> {
291                            let mut buf = [0; 1024];
292
293                            let mut header_bytes = Vec::new();
294                            let mut packet: ApiPacket = ApiPacket::default();
295
296                            // Read data from socket
297                            loop {
298                                let n = match reader.read(&mut buf).await {
299                                    // socket closed
300                                    Ok(0) => return Ok(()),
301                                    Ok(n) => n,
302                                    Err(e) => return Err(e.into()),
303                                };
304                                for byte in &buf[0..n] {
305                                    if header_bytes.len() < 4 {
306                                        header_bytes.push(*byte);
307                                        if header_bytes.len() == 4 {
308                                            if let Ok(header_success) =
309                                                DefaultOptions::new().with_fixint_encoding().with_big_endian().deserialize(&header_bytes)
310                                            {
311                                                debug!("Deserialized header: {:?}", header_success);
312                                                packet.header = header_success;
313                                            } else {
314                                                return Err("Could not deserialize header".into());
315                                            }
316                                        }
317                                    } else {
318                                        packet.parse(*byte)?;
319                                        match packet.message {
320                                            ApiPacketMessage::Put(p) => {
321                                                debug!("Received put request: {:?}", p);
322                                                let dht = dht.clone();
323                                                tokio::spawn(async move {
324                                                    process_api_put_request(dht, p).await;
325                                                });
326                                                header_bytes.clear();
327                                                packet = ApiPacket::default();
328                                            }
329                                            ApiPacketMessage::Get(g) => {
330                                                debug!("Received get request: {:?}", g);
331                                                let writer = writer.clone();
332                                                let dht = dht.clone();
333                                                tokio::spawn(async move {
334                                                    process_api_get_request(dht, &g, &writer).await;
335                                                });
336                                                header_bytes.clear();
337                                                packet = ApiPacket::default();
338                                            }
339                                            ApiPacketMessage::Shutdown => {
340                                                debug!("Received shutdown request!");
341                                                cancellation_token_clone.cancel();
342                                                return Ok(());
343                                            }
344                                            ApiPacketMessage::Unparsed(_) => {}
345                                        }
346                                    }
347                                }
348                            }
349                        };
350                        if let Err(e) = connection_result().await {
351                            warn!("Error in API connection on port {}: {}", api_address, e)
352                        }
353                    });
354                    }
355                }
356                _ = cancellation_token.cancelled() => {
357                    info!("{}: Stopped accepting new api connections.", api_address);
358                    break;
359                }
360            }
361        }
362    });
363    // Await thread spawn, to avoid EOF errors because the thread is not ready to accept messages
364    thread_start_rx.recv().await.unwrap();
365
366    handle
367}
